# PR-026: Tab title blink on alert

Branch: pr-026-tab-blink

## Scope

When any repo is in ERROR or HUNG state, the browser tab title blinks to attract attention. This works even when the tab is in background.

## Changes

### src/web/templates/base.html

Add JavaScript at the end of the body (after existing clock script):

```javascript
(function() {
    const originalTitle = document.title;
    let blinking = false;
    let blinkInterval = null;

    function checkAlerts() {
        fetch('/api/states')
            .then(r => r.json())
            .then(states => {
                const hasAlert = states.some(s =>
                    s.state === 'ERROR' || s.state === 'HUNG'
                );
                if (hasAlert && !blinking) {
                    startBlink();
                } else if (!hasAlert && blinking) {
                    stopBlink();
                }
            })
            .catch(() => {});
    }

    function startBlink() {
        blinking = true;
        let show = true;
        blinkInterval = setInterval(() => {
            document.title = show ? 'ALERT - Pipeline Orchestrator' : originalTitle;
            show = !show;
        }, 1500);
    }

    function stopBlink() {
        blinking = false;
        if (blinkInterval) {
            clearInterval(blinkInterval);
            blinkInterval = null;
        }
        document.title = originalTitle;
    }

    setInterval(checkAlerts, 10000);
    checkAlerts();
})();
```

This polls /api/states every 10 seconds (same interval as HTMX). When alerts exist, title alternates between "ALERT - Pipeline Orchestrator" and the normal title every 1.5 seconds. When alerts clear, title returns to normal.
